Cấu trúc & Flow
try

LỖI?
except

Python exceptions.py 
# try/except — bắt nhiều loại lỗi
try:
    with open("config.json") as f:
        config = json.load(f)
except FileNotFoundError:
    print("File không tồn tại, dùng config mặc định")
    config = {"url": "http://localhost"}
except json.JSONDecodeError as e:
    print(f"JSON lỗi: {e}")
finally:
    print("Luôn chạy dòng này dù có lỗi hay không")

# Custom Exception — đặt tên lỗi rõ ràng
class TestDataError(Exception):
    """Lỗi khi dữ liệu test không hợp lệ"""
    pass

def load_test_user(data):
    if "username" not in data:
        raise TestDataError(f"Thiếu field 'username' trong: {data}")
    return data["username"]

Class, Object & Inheritance
__init__ · self · attributes & methods · kế thừa
▼
Lý thuyết — Tại sao OOP quan trọng với QC?

OOP là nền tảng của Page Object Model (POM) — pattern bắt buộc khi làm automation. Mỗi trang web sẽ là một class. Khi bạn hiểu OOP, việc học POM chỉ là áp dụng.

Python oop_basics.py 
# Khai báo class
class TestCase:
    # __init__ chạy khi tạo object mới
    def __init__(self, tc_id, name, module):
        self.tc_id   = tc_id    # instance attribute
        self.name    = name
        self.module  = module
        self.status  = "PENDING"  # giá trị mặc định
        self.duration = None

    def mark_pass(self, duration=0.0):
        self.status   = "PASS"
        self.duration = duration

    def mark_fail(self, reason=""):
        self.status = "FAIL"
        self.reason = reason

    def __repr__(self):
        return f"<TC {self.tc_id} | {self.status}>"


# Tạo object (instance)
tc1 = TestCase("TC001", "Login valid user", "auth")
tc1.mark_pass(1.5)
print(tc1)   # <TC TC001 | PASS>


# Kế thừa (Inheritance)
class TestSuite(TestCase):   # kế thừa từ TestCase
    def __init__(self, suite_id, name):
        super().__init__(suite_id, name, module="suite")
        self.test_cases = []

    def add_test(self, tc: TestCase):
        self.test_cases.append(tc)

    def report(self):
        total  = len(self.test_cases)
        passed = sum(1 for t in self.test_cases if t.status == "PASS")
        print(f"Suite: {self.name} — {passed}/{total} passed")
self là gì — giải thích thật đơn giản

Đây là câu hỏi mà hầu hết người mới đều thắc mắc. Python không tự biết method đang chạy cho object nào — nên bạn phải truyền object đó vào thủ công. Tên self chỉ là quy ước, bạn đặt tên khác cũng được (nhưng đừng làm vậy).

Python self_explained.py 
# Khi bạn gọi:
tc1.mark_pass(1.5)

# Python ngầm dịch thành:
TestCase.mark_pass(tc1, 1.5)
# → self chính là tc1 bên trong method!

# Nếu không có self, method không biết
# đang chỉnh status của tc1 hay tc2:
tc1 = TestCase("TC001", "Login", "auth")
tc2 = TestCase("TC002", "Logout", "auth")

tc1.mark_pass()   # self = tc1  → chỉ tc1.status đổi
tc2.mark_fail()   # self = tc2  → chỉ tc2.status đổi

print(tc1.status)  # PASS
print(tc2.status)  # FAIL
💡
Quy tắc nhớ: Mỗi method muốn truy cập data của chính object đó thì phải có self làm tham số đầu tiên. Không có self → method không biết mình đang làm việc với object nào.
Class attribute vs Instance attribute

Hai loại attribute này hoạt động rất khác nhau — nhầm lẫn chỗ này sẽ gây bug khó tìm. Class attribute được chia sẻ chung cho tất cả object. Instance attribute thuộc riêng từng object, khai báo bên trong __init__ qua self.

Python class_vs_instance_attr.py 
class TestCase:
    # Class attribute — dùng chung, khai báo ngoài __init__
    category = "functional"
    count    = 0   # đếm tổng số TC đã tạo

    def __init__(self, tc_id, name):
        # Instance attribute — riêng của từng object
        self.tc_id  = tc_id
        self.name   = name
        self.status = "PENDING"
        TestCase.count += 1   # tăng bộ đếm chung


tc1 = TestCase("TC001", "Login")
tc2 = TestCase("TC002", "Logout")

# Class attribute — truy cập qua tên class hoặc object
print(TestCase.category)   # functional
print(tc1.category)        # functional (kế thừa từ class)
print(TestCase.count)      # 2  (tc1 + tc2)

# Instance attribute — riêng từng object
print(tc1.tc_id)            # TC001
print(tc2.tc_id)            # TC002

# ⚠️ Cẩn thận: gán qua object tạo instance attr mới,
#    không đổi class attr!
tc1.category = "regression"   # tạo instance attr riêng cho tc1
print(tc1.category)            # regression (instance attr)
print(tc2.category)            # functional  (class attr gốc)
print(TestCase.category)      # functional  (không đổi)
⚠️
Dễ nhầm: Nếu class attribute là mutable (list, dict), tất cả object dùng chung cùng một object đó. Thay đổi từ một object → ảnh hưởng hết. Luôn khai báo list/dict mặc định bên trong __init__ qua self để tránh lỗi này.
Kế thừa — is-a hay has-a?

Kế thừa chỉ hợp lý khi có quan hệ "is-a" (là một loại của). Dùng sai sẽ khiến code khó hiểu và khó sửa. Kiểm tra nhanh: đọc câu "A là một B" — nếu câu đó nghe tự nhiên thì kế thừa đúng, nếu nghe gượng thì dùng composition (has-a) thay vào.

Python inheritance_vs_composition.py 
# ✅ is-a → dùng kế thừa hợp lý
# "AutomatedTestCase là một loại TestCase" → đúng
class AutomatedTestCase(TestCase):
    def __init__(self, tc_id, name, script_path):
        super().__init__(tc_id, name)
        self.script_path = script_path

    def run(self):
        print(f"Chạy script: {self.script_path}")
        self.mark_pass()   # dùng method kế thừa từ TestCase


# ❌ has-a → KHÔNG nên kế thừa
# "TestSuite là một loại TestCase" → nghe sai!
# TestSuite CHỨA nhiều TestCase → dùng composition
class TestSuite:   # không kế thừa TestCase
    def __init__(self, name):
        self.name       = name
        self.test_cases = []   # has-a: chứa list TestCase

    def add(self, tc: TestCase):
        self.test_cases.append(tc)

    def report(self):
        total  = len(self.test_cases)
        passed = sum(1 for t in self.test_cases if t.status == "PASS")
        print(f"[{self.name}] {passed}/{total} passed")


# Kết hợp: tạo suite, thêm các TC vào
suite = TestSuite("Smoke Test")
tc_auto = AutomatedTestCase("TC001", "Login", "tests/login.py")
tc_auto.run()
suite.add(tc_auto)
suite.report()   # [Smoke Test] 1/1 passed
