Python là ngôn ngữ dynamically typed — bạn không cần khai báo kiểu dữ liệu trước. Python tự hiểu kiểu của biến dựa vào giá trị được gán.

Python variables.py 
# Các kiểu dữ liệu cơ bản
name     = "Selenium"       # str  — chuỗi
version  = 3               # int  — số nguyên
speed    = 0.5             # float — số thực
is_pass  = True            # bool — True / False

# Kiểm tra kiểu
print(type(name))         # <class 'str'>

# f-string — cách ghép chuỗi hiện đại nhất
result = f"Test {name} v{version} — Pass: {is_pass}"
print(result)
# → Test Selenium v3 — Pass: True

# Ép kiểu (Type casting)
score_str = "95"
score_int = int(score_str)     # str → int
score_f   = float(score_str)   # str → float
💡
Quan trọng với QC: Khi đọc dữ liệu từ file CSV hay form web, mọi thứ đều là str. Bạn cần ép kiểu thủ công trước khi so sánh số.

Operators (Toán tử) — 3 loại thường dùng nhất:

Python 
# Arithmetic: + - * / // % **
total  = 10 + 5     # 15
ratio  = 7 / 2      # 3.5  (luôn ra float)
div    = 7 // 2     # 3    (floor division)
remain = 7 % 2      # 1    (phần dư)

# Comparison: == != > < >= <=  → trả về bool
print(5 == 5)       # True
print("abc" != "xyz") # True

# Logical: and  or  not
is_admin = True
is_active = False
print(is_admin and is_active)  # False
print(is_admin or is_active)   # True
print(not is_active)           # True

Python loops.py 
# for loop — lặp qua từng phần tử
test_cases = ["TC001", "TC002", "TC003"]
for tc in test_cases:
    print(f"Running {tc}")

# range() — lặp n lần
for i in range(1, 6):   # 1, 2, 3, 4, 5
    print(f"Attempt {i}")

# enumerate() — lấy cả index lẫn giá trị
for idx, tc in enumerate(test_cases, start=1):
    print(f"[{idx}] {tc}")

# while loop + break/continue
retries = 0
while retries < 3:
    retries += 1
    print(f"Retry {retries}")
    if retries == 2:
        break  # thoát ngay

Function giúp bạn tái sử dụng code và tách logic ra thành các đơn vị nhỏ có thể test độc lập — đây là nền tảng để hiểu Page Object Model sau này.

# Function cơ bản
def calculate_pass_rate(total, passed):
    if total == 0:
        return 0.0
    return round(passed / total * 100, 2)

rate = calculate_pass_rate(20, 17)  # 85.0

# Default parameter
def create_test_case(name, status="PENDING", priority="medium"):
    return {"name": name, "status": status, "priority": priority}

tc1 = create_test_case("Login test")          # dùng default
tc2 = create_test_case("Checkout", "PASS")   # override status

# *args — nhận nhiều positional arguments
def run_tests(*test_names):
    for name in test_names:
        print(f"Running: {name}")

run_tests("TC001", "TC002", "TC003")

# **kwargs — nhận nhiều keyword arguments
def log_result(**kwargs):
    for k, v in kwargs.items():
        print(f"  {k} = {v}")

log_result(test="Login", status="PASS", duration=1.23)

Trong automation testing, bạn sẽ liên tục đọc test data từ file (CSV/JSON) và ghi kết quả ra file báo cáo. Đây là kỹ năng cốt lõi.

Python file_io.py 
import csv
import json

## ĐỌC CSV ##
# test_data.csv có dạng: username,password,expected
with open("test_data.csv", "r", encoding="utf-8") as f:
    reader = csv.DictReader(f)  # mỗi row là dict
    for row in reader:
        print(row["username"], row["expected"])

## GHI JSON ##
results = [
    {"id": "TC001", "status": "PASS", "duration": 1.23},
    {"id": "TC002", "status": "FAIL", "duration": 2.01},
]
with open("report.json", "w", encoding="utf-8") as f:
    json.dump(results, f, indent=2, ensure_ascii=False)

## ĐỌC JSON ##
with open("report.json", "r") as f:
    data = json.load(f)    # trả về list/dict Python
    print(data[0]["status"])
✅
Best practice: Luôn dùng with open(...) thay vì f = open(). with tự đóng file khi xong, kể cả khi có lỗi xảy ra.

Class, Object & Inheritance
__init__ · self · attributes & methods · kế thừa
▼
Lý thuyết — Tại sao OOP quan trọng với QC?

OOP là nền tảng của Page Object Model (POM) — pattern bắt buộc khi làm automation. Mỗi trang web sẽ là một class. Khi bạn hiểu OOP, việc học POM chỉ là áp dụng.

Python oop_basics.py 
# Khai báo class
class TestCase:
    # __init__ chạy khi tạo object mới
    def __init__(self, tc_id, name, module):
        self.tc_id   = tc_id    # instance attribute
        self.name    = name
        self.module  = module
        self.status  = "PENDING"  # giá trị mặc định
        self.duration = None

    def mark_pass(self, duration=0.0):
        self.status   = "PASS"
        self.duration = duration

    def mark_fail(self, reason=""):
        self.status = "FAIL"
        self.reason = reason

    def __repr__(self):
        return f"<TC {self.tc_id} | {self.status}>"


# Tạo object (instance)
tc1 = TestCase("TC001", "Login valid user", "auth")
tc1.mark_pass(1.5)
print(tc1)   # <TC TC001 | PASS>


# Kế thừa (Inheritance)
class TestSuite(TestCase):   # kế thừa từ TestCase
    def __init__(self, suite_id, name):
        super().__init__(suite_id, name, module="suite")
        self.test_cases = []

    def add_test(self, tc: TestCase):
        self.test_cases.append(tc)

    def report(self):
        total  = len(self.test_cases)
        passed = sum(1 for t in self.test_cases if t.status == "PASS")
        print(f"Suite: {self.name} — {passed}/{total} passed")
