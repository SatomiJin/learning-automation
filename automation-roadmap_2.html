score_f   = float(score_str)   # str → float

Class, Object & Inheritance
__init__ · self · attributes & methods · kế thừa
▼
Lý thuyết — Tại sao OOP quan trọng với QC?

OOP là nền tảng của Page Object Model (POM) — pattern bắt buộc khi làm automation. Mỗi trang web sẽ là một class. Khi bạn hiểu OOP, việc học POM chỉ là áp dụng.

Python oop_basics.py 
# Khai báo class
class TestCase:
    # __init__ chạy khi tạo object mới
    def __init__(self, tc_id, name, module):
        self.tc_id   = tc_id    # instance attribute
        self.name    = name
        self.module  = module
        self.status  = "PENDING"  # giá trị mặc định
        self.duration = None

    def mark_pass(self, duration=0.0):
        self.status   = "PASS"
        self.duration = duration

    def mark_fail(self, reason=""):
        self.status = "FAIL"
        self.reason = reason

    def __repr__(self):
        return f"<TC {self.tc_id} | {self.status}>"


# Tạo object (instance)
tc1 = TestCase("TC001", "Login valid user", "auth")
tc1.mark_pass(1.5)
print(tc1)   # <TC TC001 | PASS>


# Kế thừa (Inheritance)
class TestSuite(TestCase):   # kế thừa từ TestCase
    def __init__(self, suite_id, name):
        super().__init__(suite_id, name, module="suite")
        self.test_cases = []

    def add_test(self, tc: TestCase):
        self.test_cases.append(tc)

    def report(self):
        total  = len(self.test_cases)
        passed = sum(1 for t in self.test_cases if t.status == "PASS")
        print(f"Suite: {self.name} — {passed}/{total} passed")

from playwright.sync_api import sync_playwright

with sync_playwright() as p:

Assertions với expect()
to_be_visible · to_have_text · to_have_url · to_be_enabled
▼
Lý thuyết

Assertion là trái tim của mọi test case — đây là nơi bạn xác nhận rằng ứng dụng đang hoạt động đúng. Playwright cung cấp expect() với auto-retry tích hợp sẵn.

Python assertions.py 
from playwright.sync_api import expect

## Page assertions
expect(page).to_have_url("https://example.com/dashboard")
expect(page).to_have_title("Dashboard")

## Locator assertions — TỰ RETRY cho đến khi pass hoặc timeout
expect(page.locator("h1")).to_be_visible()
expect(page.locator(".error-msg")).to_have_text("Invalid credentials")
expect(page.locator("#submit")).to_be_enabled()
expect(page.locator("#submit")).to_be_disabled()
expect(page.locator("input#email")).to_have_value("[email protected]")
expect(page.locator(".cart-badge")).to_have_text("3")
expect(page.locator("#agree")).to_be_checked()

## NOT — phủ định assertion
expect(page.locator(".error")).not_to_be_visible()

## Custom timeout
expect(page.locator(".spinner")).not_to_be_visible(timeout=10_000)  # 10s

## Soft assertion — ghi nhớ lỗi, không dừng test ngay
from playwright.sync_api import expect
expect.soft(page.locator("h1")).to_have_text("Welcome")
expect.soft(page.locator(".subtitle")).to_be_visible()
page.context.tracing.stop()  # báo lỗi cuối cùng
⚠️
Tránh dùng assert element.is_visible() — cách này không có retry. Luôn dùng expect(locator).to_be_visible() để test ổn định hơn.

Waits & Network handling
Auto-waiting · wait_for_load_state · wait_for_response · page.route()
▼
Lý thuyết — Auto-waiting là gì?

Playwright tự động chờ element trước khi thực hiện action. Khi bạn click(), Playwright chờ element: visible, stable (không đang animate), enabled. Đây là lý do Playwright ổn định hơn Selenium.

Python waits.py 
## TUYỆT ĐỐI TRÁNH — time.sleep() cứng nhắc
import time
time.sleep(3)  # ❌ KHÔNG BAO GIỜ làm thế này!

## ĐÚNG — chờ theo trạng thái trang
page.wait_for_load_state("networkidle")  # chờ hết request
page.wait_for_load_state("domcontentloaded")
page.wait_for_load_state("load")

## Chờ element xuất hiện
page.locator(".result-list").wait_for(state="visible")
page.locator(".spinner").wait_for(state="hidden")  # chờ spinner biến mất

## Chờ navigation sau khi click
with page.expect_navigation():
    page.get_by_role("button", name="Submit").click()

## Mock API — rất hữu ích để test error states!
def mock_error(route):
    route.fulfill(status=500, body='{"error": "Server down"}')

page.route("**/api/products", mock_error)
page.goto("/products")
# Page sẽ nhận response 500, test UI xử lý lỗi

## Intercept response — kiểm tra API call
with page.expect_response("**/api/login") as response_info:
    page.get_by_role("button", name="Login").click()
response = response_info.value
print(response.status)  # 200

## DIALOGS (alert / confirm / prompt) ##
# Playwright auto-dismiss dialogs — cần handle thủ công
def handle_dialog(dialog):
    print(f"Dialog type: {dialog.type}, message: {dialog.message}")
    if dialog.type == "confirm":
        dialog.accept()    # nhấn OK
    else:
        dialog.dismiss()   # nhấn Cancel/close

page.on("dialog", handle_dialog)
page.get_by_role("button", name="Delete").click()
# dialog.accept() tự động được gọi

## iFRAMES ##
# Lấy frame theo URL hoặc name

# Bắt sự kiện khi trang mở tab mới
with context.expect_page() as new_page_info:

Tuần 9 · Page Object Model (POM)
🏛️
Page Object Model — Kiến trúc POM
BasePage · locators tập trung · action methods · tách logic
▼
Lý thuyết — POM là gì và tại sao cần dùng?

POM (Page Object Model) là design pattern tổ chức code automation. Thay vì viết locator và action rải rác trong test, bạn tạo một class cho mỗi trang — class đó chứa locators và các method tương tác.

💡
Lợi ích POM: Khi UI thay đổi (ví dụ dev đổi id của nút), bạn chỉ cần sửa 1 chỗ trong page class — thay vì tìm và sửa trong 20 file test.

Python pages/base_page.py 
from playwright.sync_api import Page, expect

class BasePage:
    """Class cha chứa các method dùng chung"""

    def __init__(self, page: Page):
        self.page = page

    def navigate(self, url: str):
        self.page.goto(url)

    def wait_for_page_load(self):
        self.page.wait_for_load_state("networkidle")

    def take_screenshot(self, name: str):
        self.page.screenshot(path=f"screenshots/{name}.png")

    def get_title(self) -> str:
        return self.page.title()
Python pages/login_page.py 
from playwright.sync_api import expect
from pages.base_page import BasePage

class LoginPage(BasePage):
    # Locators — tập trung tại đây
    URL          = "https://www.saucedemo.com"
    USERNAME_INP = "#user-name"
    PASSWORD_INP = "#password"
    LOGIN_BTN    = "#login-button"
    ERROR_MSG    = "[data-test='error']"

    def open(self):
        self.navigate(self.URL)
        return self

    def login(self, username: str, password: str):
        self.page.fill(self.USERNAME_INP, username)
        self.page.fill(self.PASSWORD_INP, password)
        self.page.click(self.LOGIN_BTN)
        return self  # cho phép method chaining

    def get_error_message(self) -> str:
        return self.page.locator(self.ERROR_MSG).inner_text()

    def assert_error_visible(self):
        expect(self.page.locator(self.ERROR_MSG)).to_be_visible()
        return self

Best Practices — Viết test như senior

Những điều KHÔNG BAO GIỜ làm

Danh sách đen:
• time.sleep() — dùng expect().to_be_visible() thay thế

• Test phụ thuộc nhau (test B phải chạy sau test A)

• Mỗi test độc lập, chạy được riêng lẻ
• Tên test rõ ràng: test_login_with_invalid_password_shows_error
• Dùng @pytest.fixture cho setup/teardown

• Comment giải thích tại sao chứ không phải làm gì
